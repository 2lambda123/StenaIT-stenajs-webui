## Install

```bash
npm install --save stenajs-webui-common
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'stenajs-webui-common'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
